#!/bin/bash  
  
# Step 1: Create a new conda environment with Python 3.9  
conda create --name huggingface1 python==3.9 --yes  
  
# Step 2: Activate the new environment  
source activate huggingface1  
  
# Step 3: Install notebook and ipykernel  
conda install notebook ipykernel --yes  
  
# Step 4: Configure the Jupyter kernel to use the new environment  
ipython kernel install --name huggingface1 --display-name "Python 3.9 (huggingface1)" --user  
  
# Step 5: Install dependencies from requirements.txt using pip  
pip install -r requirements.txt  
  
echo "Environment setup is complete."
